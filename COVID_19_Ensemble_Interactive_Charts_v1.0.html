fig, ax = plt.subplots(1, figsize=(8,5))

def update_plot(input_1, input_2, input_3):
    
    ax.clear()
    
    y = stats_dict[input_2][input_1][input_3]
    
    name1 = stats_dict[input_2]['name']
    lgnd = f'{input_3} of\n{name1}'
    ax.set_title(f'{input_3} of {name1} \nfor {input_1}')
    ax.set_xlabel('$day$')
    ax.set_ylabel('Total per 10,000 Population')
    ax.plot(y, label=f'{input_3} of\n{name1}')
    ax.legend()
    plt.show()

beh_mod_wid = widgets.Dropdown(options=stats_dict['S'].keys(), value='conf_trigger_5', description='Max Behavioral Modifier')
result_wid = widgets.Dropdown(options=stats_dict.keys(), value='S', description='Model Result Type')
stat_wid = widgets.Dropdown(options=stats_dict['S']['conf_trigger_5'].keys(), value='mean', description='Stat Type')

widgets.interactive(update_plot, input_1=beh_mod_wid, input_2=result_wid, input_3=stat_wid)

fig, bx = plt.subplots(1, figsize=(8,5))

def update_plot_2(input_1, input_2, input_3):
    
    bx.clear()
    for inp in input_1:
        
        y = stats_dict[input_2][inp][input_3]
               
        name2 = stats_dict[input_2]['name']
        bx.set_title(f'{input_3} of {name2}\nfor {input_1}')
        lgnd = f'{input_3} of\n{name2}'
        bx.set_xlabel('$day$')
        bx.set_ylabel('Total per 10,000 Population')
        lgnd_set = plt.gca().get_legend_handles_labels()[1]
        bx.plot(y, label= inp if inp not in lgnd_set else '')
        bx.legend(loc=1)

        plt.show()

beh_mod_wid = widgets.SelectMultiple(options=stats_dict['S'].keys(), value=['conf_trigger_5'], description='Max Behavioral Modifier')
result_wid = widgets.ToggleButtons(options=stats_dict.keys(), disabled=False, description='Model Result Type', button_style='success')

                                 disabled=False, description='Stat Type', button_style='info')


widgets.interactive(update_plot_2, input_1=beh_mod_wid, input_2=result_wid, input_3=stat_wid)
