fig, ax = plt.subplots(1, figsize=(4.5,4))
plt.suptitle('Model Results with Widgets')

def update_plot(input_1, input_2, input_3):
    
    ax.clear()
    y = stats_dict[input_2][input_1][input_3]
    name1 = stats_dict[input_2]['name']
#     name2 = stats_dict[input_1]['name']
    plt.suptitle(f'{input_3} of {name1} for {input_1}')
    
    ax.plot(y)
    plt.show()

# beh_mod_wid_1 = widgets.IntSlider(min=0, max=999, value=0, description='Model Run')

stat_wid = widgets.Dropdown(options=stats_dict['S']['conf_trigger_5'].keys(), value='mean', description='Stat Type')


widgets.interactive(update_plot, input_1=beh_mod_wid, input_2=result_wid, input_3=stat_wid)

def update_plot_2(input_1, input_2, input_3):
